CREATE TABLE app_user(
  id integer PRIMARY KEY,
  name VARCHAR(50)
);

CREATE TABLE budget(
  id integer PRIMARY KEY,
  name VARCHAR(50),
  balance money
  app_user_id integer REFERENCES app_user(id)
);

CREATE TABLE transaction(
  id integer PRIMARY KEY,
  date date,
  amount money,
  recipient varchar(50),
  budget_id integer REFERENCES budget(id)
);

ALTER TABLE app_user ALTER COLUMN id ADD GENERATED BY DEFAULT AS IDENTITY;

ALTER TABLE transaction ALTER COLUMN id ADD GENERATED BY DEFAULT AS IDENTITY;
